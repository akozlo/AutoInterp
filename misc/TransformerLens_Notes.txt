'''
Available activation tensors for layer n:
hook_embed  == embedding layer
hook_pos_embed  == position embedding layer
blocks.n.hook_resid_pre  == resid_pre.std(dim=-1, keepdim=True, unbiased=False)
blocks.n.ln1.hook_scale  == std of the residual stream before the attention layer
blocks.n.ln1.hook_normalized  == normalized residual connection before the attention layer
blocks.n.attn.hook_q  == query activation tensor
blocks.n.attn.hook_k  == key activation tensor
blocks.n.attn.hook_v  == value activation tensor
blocks.n.attn.hook_attn_scores  == attention scores (before softmax) tensor
blocks.n.attn.hook_pattern  == attention pattern (after softmax) tensor
blocks.n.attn.hook_z  == product of attention pattern and value tensor
blocks.n.hook_attn_out  == output of attention layer (attn.hook_z %% W_O)
blocks.n.hook_resid_mid  == residual stream right before the MLP
blocks.n.ln2.hook_scale  == resid_mid.std(dim=-1, keepdim=True, unbiased=False)
blocks.n.ln2.hook_normalized  == normalized residual stream right before MLP
blocks.n.mlp.hook_pre  == activations before non-linear activation in MLP
blocks.n.mlp.hook_post  == activations after non-linear activation in MLP
blocks.n.hook_mlp_out == actvation after mlp_out (mlp.hook_post %% mlp.W_out)
blocks.n.hook_resid_post  == residual stream right after the MLP
ln_final.hook_scale == resid_pre.std(dim=-1, keepdim=True, unbiased=False)
ln_final.hook_normalized == normalized residual stream right before unembedding

'''
